<objective>
Wrap the close position operation in a Drift database transaction to guarantee atomicity.

Purpose: Fix Pitfall 1 (non-atomic close) — currently `closePosition` deletes the open position then inserts the closed position as two separate operations. If the app crashes between them, the open position is lost with no closed position created. This is a critical data integrity bug.

Output: Atomic closePosition that uses Drift's `transaction()` to wrap delete+insert in a single database transaction.
</objective>

    ```dart
    /// Atomically close an open position: delete open + insert closed in one transaction.
    /// Returns the raw closed position data map, or null if open position not found.
    Future<Map<String, dynamic>?> closePositionInTransaction({
      required String openPositionId,
      required String userId,
      required double closePrice,
      required DateTime closeTime,
      required String reason, // CloseReason.name
      required double swap,
      required double commission,
      required double profit,
      required DateTime openTime,
      required DateTime createdAt,
      required String symbol,
      required double volume,
      required String side,
      required double openPrice,
      required double? stopLoss,
      required double? takeProfit,
    }) async {
      return transaction(() async {
        // 1. Verify open position exists
        final openData = await getOpenPositionById(openPositionId);
        if (openData == null) return null;

        // 2. Delete open position
        await (delete(openPositions)..where((t) => t.id.equals(openPositionId))).go();

        // 3. Insert closed position
        final closedId = openPositionId; // preserve same ID
        await into(closedPositions).insert(ClosedPositionsCompanion(
          id: Value(closedId),
          userId: Value(userId),
          symbol: Value(symbol),
          openTime: Value(openTime),
          closeTime: Value(closeTime),
          volume: Value(volume),
          side: Value(side),
          openPrice: Value(openPrice),
          closePrice: Value(closePrice),
          stopLoss: stopLoss != null ? Value(stopLoss) : const Value.absent(),
          takeProfit: takeProfit != null ? Value(takeProfit) : const Value.absent(),
          swap: Value(swap),
          commission: Value(commission),
          profit: Value(profit),
          reason: Value(reason),
          createdAt: Value(createdAt),
          updatedAt: Value(DateTime.now()),
          isSynced: const Value(false),
        ));

        // 4. Return the closed position data
        return {
          'id': closedId,
          'userId': userId,
          'symbol': symbol,
          'openTime': openTime.toIso8601String(),
          'closeTime': closeTime.toIso8601String(),
          'volume': volume,
          'side': side,
          'openPrice': openPrice,
          'closePrice': closePrice,
          'stopLoss': stopLoss,
          'takeProfit': takeProfit,
          'swap': swap,

          'profit': profit,
          'reason': reason,
          'createdAt': createdAt.toIso8601String(),
          'updatedAt': DateTime.now().toIso8601String(),
          'isSynced': false,
        };
      });
    }

    @override
    Future<Result<ClosedPosition>> closePosition({
      required String openPositionId,
      required double closePrice,
      required DateTime closeTime,
      required CloseReason reason,
    }) async {
      try {
        // 1. Fetch open position first to get all required fields
        final openDataMap = await _localDataSource.getOpenPositionById(openPositionId);
        if (openDataMap == null) {
          return const Result.failure('Open position not found');
        }

        final openDto = OpenPositionDto.fromJson(openDataMap);
        final openPosition = openDto.toEntity();

        // 2. Calculate profit
        final profit = openPosition.side.calculateProfit(
          openPosition.openPrice,
          closePrice,
          openPosition.volume,
        );

        // 3. Execute atomic close (delete open + insert closed in single transaction)
        final closedDataMap = await _localDataSource.closePositionInTransaction(
          openPositionId: openPositionId,
          userId: openPosition.userId,
          closePrice: closePrice,
          closeTime: closeTime,
          reason: reason.name,
          swap: openPosition.swap,
          commission: openPosition.commission,
          profit: profit,
          openTime: openPosition.openTime,
          createdAt: openPosition.createdAt,
          symbol: openPosition.symbol,
          volume: openPosition.volume,
          side: openPosition.side.name,
          openPrice: openPosition.openPrice,
          stopLoss: openPosition.stopLoss,
          takeProfit: openPosition.takeProfit,
        );

        if (closedDataMap == null) {
          return const Result.failure('Open position not found');
        }

        // 4. Convert result back to ClosedPosition entity
        final closedDto = ClosedPositionDto.fromJson(closedDataMap);
        return Result.success(closedDto.toEntity());
      } catch (e) {
        return Result.failure('Failed to close position: $e');
      }
    }
    ```

    This replaces the old separate `_localDataSource.deleteOpenPosition` + `_localDataSource.insertClosedPosition` calls with a single atomic `_localDataSource.closePositionInTransaction` call.

    **Important:** The import for `OpenPositionDto` and `ClosedPositionDto` is already present in the file. The `TradeSide.calculateProfit` is accessed via `openPosition.side.calculateProfit(...)` which already works.
